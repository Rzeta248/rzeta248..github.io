#define IN1 7   // Control direcci√≥n motor A

// Bluetooth
#define BT_RX 10
#define BT_TX 11

SoftwareSerial BT(BT_RX, BT_TX);

int command;
int Speed = 204; // 0-255

void setup() {
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(LEDFA, OUTPUT);
  pinMode(LEDB, OUTPUT);
  pinMode(HORN, OUTPUT);
  
  pinMode(LEDUP, OUTPUT);

  Serial.begin(9600);
  BT.begin(9600);
  Serial.println("Sistema iniciado...");

void loop() {
  prenderled();
  

  if (BT.available() > 0) {
    command = BT.read();
    Serial.print("Comando: ");
    Serial.println(char(command));
    handleCommand(command);
  }

  if (Serial.available() > 0) {

void handleCommand(int cmd) {
  switch (cmd) {
    case 'F': forward(); break;
    case 'B': back(); break;
    case 'L': left(); break;
    case 'R': right(); break;
    case 'G': forwardleft(); break;
    case 'I': forwardright(); break;
    case 'H': backleft(); break;
    case 'J': backright(); break;
    case 'S': Stop(); break;
    case 'W': frontledon(); break;
    case 'w': frontledoff(); break;
    case 'U': backledon(); break;
    case 'u': backledoff(); break;
    case 'V': hornon(); break;
    case 'v': hornoff(); break;
    case '0': Speed = 100; break;
    case '1': Speed = 140; break;
    case '2': Speed = 153; break;
    case '3': Speed = 165; break;
    case '4': Speed = 178; break;
    case '5': Speed = 191; break;
    case '6': Speed = 204; break;
    case '7': Speed = 216; break;
    case '8': Speed = 229; break;
    case '9': Speed = 242; break;
    case 'q': Speed = 255; break;
    default: Stop(); break;
  }
}

void setMotorSpeed() {
  analogWrite(ENA, Speed);
  analogWrite(ENB, Speed);
}

void forward() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
  setMotorSpeed();
}

void back() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  setMotorSpeed();
}

void left() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
  setMotorSpeed();
}

void right() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  setMotorSpeed();
}

void forwardleft() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  analogWrite(ENA, Speed/2); // Reducir velocidad motor izquierdo
  analogWrite(ENB, Speed);
}

void forwardright() {
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  analogWrite(ENA, Speed);
  analogWrite(ENB, Speed/2); // Reducir velocidad motor derecho
}

void backleft() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
  analogWrite(ENA, Speed/2);
  analogWrite(ENB, Speed);
}

void backright() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
  analogWrite(ENA, Speed);
  analogWrite(ENB, Speed/2);
}

void Stop() {
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);
}
